## 1. IM系统概述

### 1.1 核心功能

```
单聊: 一对一文字/图片/语音消息
群聊: 多人多消息
```

### 2.2 Netty服务端实现

```java
/**
 * IM连接服务器 — 基于Netty
 * 单机支撑10万级长连接
 */
@Component
public class IMServer {

    private EventLoopGroup bossGroup;
    private EventLoopGroup workerGroup;
    private Channel serverChannel;

    public void start(int port) {
        // 使用Epoll (Linux性能更好)
        bossGroup = new EpollEventLoopGroup(1);
        workerGroup = new EpollEventLoopGroup();

        ServerBootstrap bootstrap = new ServerBootstrap();
        bootstrap.group(bossGroup, workerGroup)
            .channel(EpollServerSocketChannel.class)
            .option(ChannelOption.SO_BACKLOG, 1024)
            .option(ChannelOption.SO_REUSEADDR, true)
            .childOption(ChannelOption.TCP_NODELAY, true)
            .childOption(ChannelOption.SO_KEEPALIVE, true)
            .childOption(ChannelOption.ALLOCATOR, PooledByteBufAllocator.DEFAULT)
            .childHandler(new ChannelInitializer<SocketChannel>() {
                @Override
                protected void initChannel(SocketChannel ch) {
                    ChannelPipeline pipeline = ch.pipeline();

                    // 空闲检测: 60秒无读事件则关闭连接
                    pipeline.addLast(new IdleStateHandler(60, 0, 0));

                    // 心跳处理
                    pipeline.addLast(new HeartbeatHandler());

                    // 消息编解码: 长度字段 + 消息体
                    pipeline.addLast(new LengthFieldBasedFrameDecoder(
                        1024 * 1024,  // 最大帧长度1MB
                        0,             // 长度字段偏移
                        4,             // 长度字段长度
                        0,             // 长度调整值
                        4              // 跳过的初始字节
                    ));
                    pipeline.addLast(new LengthFieldPrepender(4));

                    // 消息序列化/反序列化 (Protobuf)
                    pipeline.addLast(new ProtobufDecoder(IMMessage.ProtocolMessage.getDefaultInstance()));
                    pipeline.addLast(new ProtobufEncoder());

                    // 业务处理
                    pipeline.addLast(new IMMessageHandler());
                }
            });

        try {
            ChannelFuture future = bootstrap.bind(port).sync();
            serverChannel = future.channel();
            log.info("IM Server started on port {}", port);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new RuntimeException("IM Server启动失败", e);
        }
    }

    public void shutdown() {
        if (serverChannel != null) serverChannel.close();
        if (bossGroup != null) bossGroup.shutdownGracefully();
        if (workerGroup != null) workerGroup.shutdownGracefully();
    }
}
```

### 2.3 连接管理

```java
/**
 * 连接管理器
 * 维护用户ID与Channel的映射关系
 */
@Component
public class ConnectionManager {

    // 本机连接表: userId -> Channel
    private final ConcurrentHashMap<Long, Channel> userChannels = new ConcurrentHashMap<>();

    // Channel属性键
    private static final AttributeKey<Long> USER_ID_KEY = AttributeKey.valueOf("userId");

    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    /**
     * 注册用户连接
     */
    public void registerConnection(Long userId, Channel channel) {
        // 绑定用户ID到Channel属性
        channel.attr(USER_ID_KEY).set(userId);

        // 存入本机连接表
        Channel oldChannel = userChannels.put(userId, channel);
        if (oldChannel != null && oldChannel != channel) {
            // 同一用户在其他地方登录，踢掉旧连接
            oldChannel.close();
        }

        // 注册到Redis，记录用户连接在哪台机器上
        String connServer = getLocalServerId();
        redisTemplate.opsForValue().set("im:conn:" + userId, connServer, 300, TimeUnit.SECONDS);

        // 更新在线状态
        redisTemplate.opsForSet().add("im:online:" + connServer, String.valueOf(userId));

        log.info("用户 {} 连接注册到服务器 {}", userId, connServer);
    }

    /**
     * 移除用户连接
     */
    public void removeConnection(Long userId) {
        userChannels.remove(userId);
        redisTemplate.delete("im:conn:" + userId);
        redisTemplate.opsForSet().remove("im:online:" + getLocalServerId(), String.valueOf(userId));
    }

    /**
     * 查找用户连接
     */
    public Channel getConnection(Long userId) {
        return userChannels.get(userId);
    }

    /**
     * 判断用户是否在线
     */
    public boolean isOnline(Long userId) {
        return redisTemplate.hasKey("im:conn:" + userId);
    }

    /**
     * 获取用户连接所在的服务器ID
     */
    public String getConnServer(Long userId) {
        return redisTemplate.opsForValue().get("im:conn:" + userId);
    }
}
```

### 2.4 心跳保活

```java
/**
 * 心跳处理器
 * 客户端每30秒发送一次心跳，服务端90秒无心跳则断开
 */
@ChannelHandler.Sharable
public class HeartbeatHandler extends ChannelInboundHandlerAdapter {

    @Override
    public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
        if (evt instanceof IdleStateEvent) {
            IdleStateEvent event = (IdleStateEvent) evt;
            if (event.state() == IdleState.READER_IDLE) {
                // 60秒未收到客户端数据，关闭连接
                Long userId = ctx.channel().attr(USER_ID_KEY).get();
                log.warn("用户 {} 心跳超时，关闭连接", userId);
                ctx.channel().close();
            }
        } else {
            super.userEventTriggered(ctx, evt);
        }
    }

    @Override
    public void channelRead(ChannelHandlerContext ctx, Object msg) {
        IMMessage.ProtocolMessage message = (IMMessage.ProtocolMessage) msg;
        if (message.getType() == IMMessage.ProtocolMessage.MessageType.HEARTBEAT) {
            // 心跳响应
            IMMessage.ProtocolMessage ack = IMMessage.ProtocolMessage.newBuilder()
                .setType(IMMessage.ProtocolMessage.MessageType.HEARTBEAT_ACK)
                .setTimestamp(System.currentTimeMillis())
                .build();
            ctx.writeAndFlush(ack);
        } else {
            ctx.fireChannelRead(msg);
        }
    }
}
```

### 3.2 消息序列号机制

```java
/**
 * 消息序列号生成器
 * 每个会话维护一个递增序列号，保证消息有序
 */
@Service
public class SequenceService {

    @Autowired
    private StringRedisTemplate redisTemplate;

    /**
     * 获取会话的下一个序列号
     * 使用Redis INCR保证原子递增
     */
    public long nextSequence(Long userId1, Long userId2) {
        // 会话Key: 固定小的userId在前，大的在后
        String sessionKey = buildSessionKey(userId1, userId2);
        String seqKey = "im:seq:" + sessionKey;
        return redisTemplate.opsForValue().increment(seqKey);
    }

    /**
     * 群消息序列号
     */
    public long nextGroupSequence(String groupId) {
        String seqKey = "im:group:seq:" + groupId;
        return redisTemplate.opsForValue().increment(seqKey);
    }

    /**
     * 构建会话Key
     * 确保同一对话的两个用户生成的Key一致
     */
    private String buildSessionKey(Long uid1, Long uid2) {
        long min = Math.min(uid1, uid2);
        long max = Math.max(uid1, uid2);
        return min + "_" + max;
    }
}
```

### 4.3 消息存储流程

```java
@Service
public class MessageStoreService {

    @Autowired
    private MessageMapper messageMapper;

    @Autowired
    private SequenceService sequenceService;

    /**
     * 存储单聊消息
     */
    public IMMessage storeMessage(SendMessageRequest request) {
        // 1. 生成消息ID
        String messageId = UUID.randomUUID().toString().replace("-", "");

        // 2. 获取会话序列号
        long sequenceId = sequenceService.nextSequence(request.getFromUid(), request.getToUid());

        // 3. 构建消息实体
        IMMessage message = new IMMessage();
        message.setMessageId(messageId);
        message.setSessionKey(buildSessionKey(request.getFromUid(), request.getToUid()));
        message.setFromUid(request.getFromUid());
        message.setToUid(request.getToUid());
        message.setSequenceId(sequenceId);
        message.setMsgType(request.getMsgType());
        message.setContent(encryptContent(request.getContent(), request.getFromUid(), request.getToUid()));
        message.setExtraData(JSON.toJSONString(request.getExtraData()));
        message.setCreateTime(LocalDateTime.now());

        // 4. 分库分表写入(ShardingSphere路由)
        messageMapper.insert(message);

        // 5. 异步写入ES索引(用于全文搜索)
        CompletableFuture.runAsync(() -> {
            indexMessageToES(message);
        });

        return message;
    }

    /**
     * 消息内容加密
     * 端到端加密: 使用双方共享的密钥加密内容
     */
    private String encryptContent(String content, Long uid1, Long uid2) {
        String sharedKey = deriveSharedKey(uid1, uid2);
        return AESUtil.encrypt(content, sharedKey);
    }

    /**
     * 拉取历史消息
     */
    public List<IMMessage> getHistoryMessages(Long fromUid, Long toUid, Long lastSequenceId, int count) {
        String sessionKey = buildSessionKey(fromUid, toUid);
        return messageMapper.selectBySession(sessionKey, lastSequenceId, count);
    }
}
```

### 5.1 离线消息存储

```java
/**
 * 离线消息管理
 * 用户不在线时，消息暂存Redis + MySQL
 */
@Service
public class OfflineMessageService {

    @Autowired
    private StringRedisTemplate redisTemplate;

    @Autowired
    private MessageMapper messageMapper;

    // Redis离线消息队列: 每个用户一个Sorted Set
    // Score = 消息时间戳, Value = 消息JSON
    private static final String OFFLINE_KEY = "im:offline:";

    /**
     * 保存离线消息
     */
    public void saveOfflineMessage(Long userId, IMMessage message) {
        String key = OFFLINE_KEY + userId;
        String messageJson = JSON.toJSONString(message);
        double score = message.getCreateTime().toEpochSecond(ZoneOffset.UTC);

        // 写入Redis Sorted Set
        redisTemplate.opsForZSet().add(key, messageJson, score);

        // 设置过期时间(7天)
        redisTemplate.expire(key, 7, TimeUnit.DAYS);

        // 记录离线消息计数
        redisTemplate.opsForValue().increment("im:offline:count:" + userId);
    }

    /**
     * 用户上线时拉取离线消息
     */
    public List<IMMessage> fetchOfflineMessages(Long userId) {
        String key = OFFLINE_KEY + userId;
        String countStr = redisTemplate.opsForValue().get("im:offline:count:" + userId);
        int count = countStr != null ? Integer.parseInt(countStr) : 100;
        count = Math.min(count, 500); // 最多一次拉取500条

        // 从Redis中取出所有离线消息
        Set<String> messages = redisTemplate.opsForZSet().range(key, 0, count - 1);
        if (messages == null || messages.isEmpty()) {
            return Collections.emptyList();
        }

        List<IMMessage> result = new ArrayList<>();
        for (String msgJson : messages) {
            result.add(JSON.parseObject(msgJson, IMMessage.class));
        }

        // 清除已拉取的离线消息
        redisTemplate.opsForZSet().removeRange(key, 0, count - 1);
        redisTemplate.delete("im:offline:count:" + userId);

        return result;
    }

    /**
     * 消息漫游（拉取更早的历史消息）
     * 从数据库分页查询
     */
    public List<IMMessage> getRoamMessages(Long userId, Long peerUid, Long beforeSeq, int size) {
        String sessionKey = buildSessionKey(userId, peerUid);
        return messageMapper.selectRoamMessages(sessionKey, beforeSeq, size);
    }
}
```

### 5.2 离线推送

```java
/**
 * 离线消息推送服务
 * 用户不在线时通过APNs/FCM推送通知
 */
@Service
public class OfflinePushService {

    @Autowired
    private ConnectionManager connectionManager;

    @Autowired
    private DeviceTokenService deviceTokenService;

    /**
     * 发送消息时判断用户是否在线
     * 不在线则触发离线推送
     */
    public void tryPushIfOffline(Long userId, IMMessage message) {
        if (connectionManager.isOnline(userId)) {
            return; // 用户在线，不需要推送
        }

        // 获取用户设备推送Token
        List<DeviceToken> tokens = deviceTokenService.getTokens(userId);
        if (tokens.isEmpty()) {
            return;
        }

        // 构建推送内容
        PushNotification notification = new PushNotification();
        notification.setTitle(getDisplayName(message.getFromUid()));
        notification.setBody(buildPushBody(message));
        notification.setData(buildPushData(message));

        // 多设备推送
        for (DeviceToken token : tokens) {
            try {
                switch (token.getPlatform()) {
                    case IOS:
                        pushViaAPNs(token.getToken(), notification);
                        break;
                    case ANDROID:
                        pushViaFCM(token.getToken(), notification);
                        break;
                    case HUAWEI:
                        pushViaHMS(token.getToken(), notification);
                        break;
                }
            } catch (Exception e) {
                log.error("离线推送失败, userId: {}, platform: {}", userId, token.getPlatform(), e);
            }
        }
    }

    private String buildPushBody(IMMessage message) {
        switch (message.getMsgType()) {
            case TEXT:
                return message.getContent();
            case IMAGE:
                return "[图片]";
            case VOICE:
                return "[语音]";
            case VIDEO:
                return "[视频]";
            default:
                return "[消息]";
        }
    }
}
```

## 6. 消息可靠性保障

### 6.1 消息可靠性三层保障

```
可靠性保障体系:
┌──────────────────────────────────────┐
│ 1. 客户端 → 服务端 (发送可靠性)       │
│    客户端ACK + 超时重传               │
├──────────────────────────────────────┤
│ 2. 服务端存储 (存储可靠性)            │
│    同步写主库 + 异步复制从库           │
├──────────────────────────────────────┤
│ 3. 服务端 → 接收端 (投递可靠性)       │
│    服务端ACK + 离线补偿 + 序列号去重  │
└──────────────────────────────────────┘
```

### 6.2 消息ACK机制

```java
/**
 * 消息投递与ACK确认
 * 保证消息至少送达一次(At-Least-Once)
 */
@Service
public class MessageDeliveryService {

    @Autowired
    private ConnectionManager connectionManager;

    @Autowired
    private StringRedisTemplate redisTemplate;

    // 待ACK消息: messageId -> 投递信息
    private static final String ACK_WAITING = "im:ack:waiting:";

    // 已投递消息去重: userId -> 最近100条messageId
    private static final String DELIVERED = "im:delivered:";

    /**
     * 投递消息给在线用户
     */
    public void deliverMessage(Long toUserId, IMMessage message) {
        Channel channel = connectionManager.getConnection(toUserId);
        if (channel == null || !channel.isActive()) {
            // 用户不在线，存储离线消息
            offlineMessageService.saveOfflineMessage(toUserId, message);
            return;
        }

        // 去重检查: 避免重复投递
        String deliveredKey = DELIVERED + toUserId;
        if (Boolean.TRUE.equals(redisTemplate.opsForSet().isMember(deliveredKey, message.getMessageId()))) {
            return; // 已投递过，跳过
        }

        // 发送消息到客户端
        ProtocolMessage protocolMsg = buildProtocolMessage(message);
        channel.writeAndFlush(protocolMsg).addListener((ChannelFutureListener) future -> {
            if (future.isSuccess()) {
                // 标记为待ACK
                String ackKey = ACK_WAITING + message.getMessageId();
                redisTemplate.opsForValue().set(ackKey, String.valueOf(toUserId), 30, TimeUnit.SECONDS);

                // 记录已投递
                redisTemplate.opsForSet().add(deliveredKey, message.getMessageId());
                redisTemplate.expire(deliveredKey, 1, TimeUnit.HOURS);
                // 限制Set大小
                if (redisTemplate.opsForSet().size(deliveredKey) > 100) {
                    // 只保留最近100条
                    redisTemplate.opsForSet().pop(deliveredKey);
                }
            } else {
                log.warn("消息投递失败, userId: {}, messageId: {}", toUserId, message.getMessageId());
                offlineMessageService.saveOfflineMessage(toUserId, message);
            }
        });
    }

    /**
     * 处理客户端ACK
     */
    public void handleAck(Long userId, String messageId) {
        String ackKey = ACK_WAITING + messageId;
        String waitingUser = redisTemplate.opsForValue().get(ackKey);

        if (waitingUser != null && waitingUser.equals(String.valueOf(userId))) {
            // ACK匹配，删除待ACK记录
            redisTemplate.delete(ackKey);
            log.debug("消息ACK确认, userId: {}, messageId: {}", userId, messageId);
        }
    }

    /**
     * 定时重试未ACK的消息
     */
    @Scheduled(fixedRate = 10000)
    public void retryUnackedMessages() {
        // 扫描待ACK消息，超时未确认的重试投递
        Set<String> keys = redisTemplate.keys(ACK_WAITING + "*");
        if (keys == null || keys.isEmpty()) return;

        for (String key : keys) {
            String messageId = key.substring(ACK_WAITING.length());
            String userId = redisTemplate.opsForValue().get(key);

            // TTL剩余时间判断是否需要重试
            Long ttl = redisTemplate.getExpire(key);
            if (ttl != null && ttl < 20) {
                // 快要过期了，重新投递
                IMMessage message = messageService.getByMessageId(messageId);
                if (message != null) {
                    Long uid = Long.parseLong(userId);
                    deliverMessage(uid, message);
                }
            }
        }
    }
}
```

```java
@Service
public class MessageOrderService {

    /**
     * 客户端收到的消息按序列号排序
     */
    public List<IMMessage> sortMessages(List<IMMessage> messages) {
        return messages.stream()
            .sorted(Comparator.comparingLong(IMMessage::getSequenceId))
            .collect(Collectors.toList());
    }

    /**
     * 检测消息缺口(序列号不连续)
     * 如果发现缺口，客户端需要拉取缺失的消息
     */
    public List<Long> detectGaps(Long userId, List<IMMessage> messages) {
        List<Long> missingSeqs = new ArrayList<>();
        if (messages.size() < 2) return missingSeqs;

        for (int i = 1; i < messages.size(); i++) {
            long prev = messages.get(i - 1).getSequenceId();
            long curr = messages.get(i).getSequenceId();
            if (curr > prev + 1) {
                // 发现缺口
                for (long seq = prev + 1; seq < curr; seq++) {
                    missingSeqs.add(seq);
                }
            }
        }
        return missingSeqs;
    }

    /**
     * 客户端同步消息
     * 基于最后接收的序列号增量拉取
     */
    public SyncResponse syncMessages(Long userId, Long lastSyncSeq) {
        // 查询所有大于lastSyncSeq的消息
        List<IMMessage> newMessages = messageMapper.selectAfterSequence(userId, lastSyncSeq, 200);

        SyncResponse response = new SyncResponse();
        response.setMessages(newMessages);
        if (!newMessages.isEmpty()) {
            response.setLatestSeq(newMessages.get(newMessages.size() - 1).getSequenceId());
        } else {
            response.setLatestSeq(lastSyncSeq);
        }
        response.setHasMore(newMessages.size() == 200);
        return response;
    }
}
```

## 8. 群消息扩散模型

### 8.1 写扩散 vs 读扩散

| 模型 | 原理 | 优点 | 缺点 | 适用场景 |
|------|------|------|------|----------|
| 写扩散 | 发消息时写入每个成员的收件箱 | 读取简单，一条消息拉取 | 写放大严重 | 小群(<200人) |
| 读扩散 | 消息只写一份到群 Timeline | 写入高效 | 读取需要合并多个Timeline | 大群(>200人) |
| 混合模式 | 小群写扩散，大群读扩散 | 兼顾性能 | 实现复杂 | 综合场景 |

### 8.2 混合扩散模型实现

```java
@Service
public class GroupMessageService {

    private static final int WRITE_FANOUT_THRESHOLD = 200; // 200人以下写扩散

    @Autowired
    private StringRedisTemplate redisTemplate;

    @Autowired
    private RocketMQTemplate rocketMQTemplate;

    /**
     * 发送群消息
     */
    public void sendGroupMessage(String groupId, Long fromUid, String content, int msgType) {
        Group group = groupService.getById(groupId);
        int memberCount = group.getMemberCount();

        // 生成消息ID和序列号
        String messageId = UUID.randomUUID().toString().replace("-", "");
        long sequenceId = sequenceService.nextGroupSequence(groupId);

        IMMessage message = new IMMessage();
        message.setMessageId(messageId);
        message.setGroupId(groupId);
        message.setFromUid(fromUid);
        message.setSequenceId(sequenceId);
        message.setContent(content);
        message.setMsgType(msgType);
        message.setCreateTime(LocalDateTime.now());

        // 存储消息(所有群消息都存一份到群Timeline)
        messageMapper.insertGroupMessage(message);

        if (memberCount <= WRITE_FANOUT_THRESHOLD) {
            // 小群：写扩散 — 写入每个成员的收件箱
            writeFanout(message, group.getMembers());
        } else {
            // 大群：读扩散 — 只存群Timeline，成员拉取时读取
            readFanout(message, groupId);
        }
    }

    /**
     * 写扩散：为每个群成员写入收件箱
     */
    private void writeFanout(IMMessage message, List<Long> memberIds) {
        String timelineKey = "im:inbox:";

        // 批量写入每个成员的Timeline
        for (Long memberId : memberIds) {
            if (memberId.equals(message.getFromUid())) continue;

            String inboxKey = timelineKey + memberId;
            String msgJson = JSON.toJSONString(message);
            double score = message.getSequenceId();

            redisTemplate.opsForZSet().add(inboxKey, msgJson, score);
            redisTemplate.expire(inboxKey, 7, TimeUnit.DAYS);

            // 如果用户在线，直接推送
            if (connectionManager.isOnline(memberId)) {
                messageDeliveryService.deliverMessage(memberId, message);
            } else {
                // 离线用户保存离线消息
                offlineMessageService.saveOfflineMessage(memberId, message);
            }
        }
    }

    /**
     * 读扩散：只更新群Timeline
     * 成员拉取消息时从群Timeline读取
     */
    private void readFanout(IMMessage message, String groupId) {
        String groupTimelineKey = "im:group:timeline:" + groupId;
        String msgJson = JSON.toJSONString(message);
        double score = message.getSequenceId();

        // 写入群Timeline
        redisTemplate.opsForZSet().add(groupTimelineKey, msgJson, score);
        redisTemplate.expire(groupTimelineKey, 7, TimeUnit.DAYS);

        // 在线成员推送通知(只推送通知，不推送完整消息)
        List<Long> onlineMembers = groupService.getOnlineMembers(groupId);
        for (Long memberId : onlineMembers) {
            if (memberId.equals(message.getFromUid())) continue;

            // 推送通知消息
            ProtocolMessage notify = ProtocolMessage.newBuilder()
                .setType(MessageType.GROUP_CHAT)
                .setGroupId(groupId)
                .setMessageId(message.getMessageId())
                .build();
            connectionManager.getConnection(memberId).writeAndFlush(notify);
        }

        // 大群消息通过MQ异步扩散
        rocketMQTemplate.asyncSend("group-fanout-topic",
            new GroupFanoutMessage(groupId, message.getMessageId()), new SendCallback() {
                @Override
                public void onSuccess(SendResult sendResult) {}
                @Override
                public void onException(Throwable e) {
                    log.error("群消息扩散失败, groupId: {}", groupId, e);
                }
            });
    }

    /**
     * 拉取群消息(读扩散模式)
     */
    public List<IMMessage> fetchGroupMessages(Long userId, String groupId, Long lastSeq, int size) {
        String groupTimelineKey = "im:group:timeline:" + groupId;

        // 从群Timeline拉取
        Set<String> messages = redisTemplate.opsForZSet()
            .rangeByScore(groupTimelineKey, lastSeq + 1, Double.MAX_VALUE, 0, size);

        if (messages == null || messages.isEmpty()) {
            // Redis中没有，从数据库拉取
            return messageMapper.selectGroupMessages(groupId, lastSeq, size);
        }

        return messages.stream()
            .map(json -> JSON.parseObject(json, IMMessage.class))
            .collect(Collectors.toList());
    }
}
```

```java
/**
 * 消息压缩传输
 * 大于256字节的消息启用GZIP压缩
 */
public class MessageCompressor extends MessageToMessageCodec<ByteBuf, ByteBuf> {

    private static final int COMPRESS_THRESHOLD = 256;

    @Override
    protected void encode(ChannelHandlerContext ctx, ByteBuf msg, List<Object> out) {
        if (msg.readableBytes() < COMPRESS_THRESHOLD) {
            out.add(msg.retain());
            return;
        }

        // GZIP压缩
        ByteBuf compressed = ctx.alloc().buffer();
        compressed.writeByte(1); // 标记: 1=压缩, 0=未压缩
        try (OutputStream os = new GZIPOutputStream(new ByteBufOutputStream(compressed))) {
            msg.readBytes(os, msg.readableBytes());
        }
        out.add(compressed);
    }

    @Override
    protected void decode(ChannelHandlerContext ctx, ByteBuf msg, List<Object> out) throws IOException {
        byte compressed = msg.readByte();
        if (compressed == 0) {
            out.add(msg.readRetainedSlice(msg.readableBytes()));
            return;
        }

        // GZIP解压
        try (InputStream is = new GZIPInputStream(new ByteBufInputStream(msg))) {
            ByteBuf decompressed = ctx.alloc().buffer();
            decompressed.writeBytes(is, is.available());
            out.add(decompressed);
        }
    }
}
```

## 10. 面试题速查

**Q1: IM系统如何保证消息不丢不重不乱序？**

不丢：发送端ACK确认+超时重传，服务端同步落盘后才返回ACK，接收端ACK确认后才从待确认队列删除。不重：每条消息有唯一messageId，接收端用LRU缓存去重。乱序：每个会话维护递增sequenceId，接收端按seq排序，检测缺口触发补拉。

**Q2: 百万用户同时在线怎么实现？**

连接层水平扩展，单机Netty支撑10万长连接。LVS/Nginx做负载均衡，连接路由信息存储在Redis(userId→serverId)。推送消息时先查Redis找到目标用户所在服务器，通过内部RPC转发到目标服务器的Netty实例。

**Q3: 群消息用写扩散还是读扩散？**

混合模式：小群(200人以下)用写扩散，消息写入每个成员的收件箱，读取简单；大群用读扩散，消息只写一份到群Timeline，成员拉取时读取。阈值根据实际性能压测确定，核心是写扩散的写放大倍数不能太大。

**Q4: 消息怎么加密传输？**

传输层TLS/SSL加密。内容层端到端加密：双方协商共享密钥(如DH密钥交换)，消息内容AES加密后传输，服务器只存储密文。群消息使用群密钥，新成员加入时重新协商密钥。

**Q5: 离线消息怎么处理？**

用户不在线时消息暂存Redis Sorted Set(按时间排序)，设置7天TTL。用户上线时批量拉取，拉取后从Redis删除。同时触发APNs/FCM离线推送通知。超过7天的离线消息从数据库拉取(消息漫游)。

**Q6: 消息存储怎么分库分表？**

按from_uid取模分4库，按月分表。热点会话数据缓存到Redis。3个月以前的数据归档到HBase。全文搜索走Elasticsearch。查询时通过Hint指定分库路由，避免全表扫描。

**Q7: 心跳机制怎么设计？**

客户端每30秒发送心跳包，服务端60秒无读事件则关闭连接。心跳包极小(几个字节)，用IdleStateHandler实现。心跳还能携带客户端状态信息(如网络类型、电量)，用于优化推送策略。

**Q8: 多端同步怎么实现？**

每端维护一个syncKey(最后同步的序列号)。同步时带上syncKey，服务端返回大于syncKey的所有消息。不同端可以有不同的syncKey，服务端只比较序列号，不关心是哪个端。新设备首次登录时全量拉取最近N条消息。
