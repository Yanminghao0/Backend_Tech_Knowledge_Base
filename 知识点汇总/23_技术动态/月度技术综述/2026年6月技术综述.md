```java
// Dubbo 3.3 虚拟线程 + Triple Streaming
@DubboService
public class UserServiceImpl implements UserService {
    @Override
    public Flowable<User> streamUsers(UserQuery query) {
        return Flowable.create(emitter -> {
            userMapper.stream(query).forEach(user -> {
                emitter.onNext(user);
            });
            emitter.onComplete();
        }, BackpressureStrategy.BUFFER);
    }
}
```
