### 2.2 wrk Lua脚本扩展

```lua
-- post_order.lua: 压测下单接口
-- 使用方法: wrk -t4 -c200 -d60s -s post_order.lua http://order-api.example.com

-- 1. 初始化阶段
local counter = 0
local userIds = {}

function setup(thread)
    -- 每个线程初始化
    thread:set("id", counter)
    counter = counter + 1
    
    -- 从文件加载测试用户
    local file = io.open("/data/test/user_tokens.txt", "r")
    for line in file:lines() do
        table.insert(userIds, line)
    end
    file:close()
end

-- 2. 每个请求前执行：设置请求头和请求体
function request()
    local userId = userIds[math.random(#userIds)]
    local token = "Bearer " .. generateToken(userId)
    
    -- 构造请求体
    local body = string.format(
        '{"userId":%s,"productId":%d,"quantity":%d,"addressId":%d}',
        userId,
        math.random(1, 1000),
        math.random(1, 5),
        math.random(1, 100)
    )
    
    return wrk.format("POST", "/api/orders", {
        ["Content-Type"] = "application/json",
        ["Authorization"] = token,
        ["X-Request-Id"] = tostring(math.random(1, 999999))
    }, body)
end

-- 3. 收到响应后执行
function response(status, headers, body)
    if status ~= 200 then
        io.write("Error: status=" .. status .. " body=" .. body .. "\n")
    end
end

-- 4. 测试结束统计
function done(summary, latency, requests)
    io.write("--- 压测结果统计 ---\n")
    io.write(string.format("总请求数: %d\n", summary.requests))
    io.write(string.format("总耗时: %.2fs\n", summary.duration / 1000000))
    io.write(string.format("QPS: %.2f\n", summary.requests / (summary.duration / 1000000)))
    io.write(string.format("错误数: %d\n", summary.errors.status))
    io.write(string.format("连接错误: %d\n", summary.errors.connect))
    io.write(string.format("读错误: %d\n", summary.errors.read))
    io.write(string.format("超时: %d\n", summary.errors.timeout))
    
    -- 延迟分布
    io.write("\n--- 延迟分布 ---\n")
    for _, p in ipairs({50, 75, 90, 99}) do
        local n = latency:percentile(p)
        io.write(string.format("P%d: %.2fms\n", p, n / 1000))
    end
end

-- 辅助函数：生成简单token（实际应从预置文件加载）
function generateToken(userId)
    return "token_" .. userId .. "_" .. os.time()
end
```

## 性能指标解读

### 3.1 核心指标详解

```
┌─────────────────────────────────────────────────────────────────┐
│                      性能指标解读                                │
├──────────────┬──────────────────────────────────────────────────┤
│ TPS          │ Transactions Per Second                         │
│ (每秒事务数) │ 每秒完成的事务数（一个事务可能包含多个请求）       │
│              │ TPS = 总事务数 / 总耗时                           │
│              │ 衡量系统的整体处理能力                            │
├──────────────┼──────────────────────────────────────────────────┤
│ QPS          │ Queries Per Second                              │
│ (每秒查询数) │ 每秒完成的请求数                                 │
│              │ QPS = 总请求数 / 总耗时                           │
│              │ 衡量接口的吞吐能力                               │
├──────────────┼──────────────────────────────────────────────────┤
│ RT           │ Response Time                                   │
│ (响应时间)   │ 单次请求从发送到收到响应的时间                    │
│              │ 关注P99而非平均值（平均值可能掩盖问题）           │
│              │ P99 = 99%的请求在此时间内完成                    │
│              │ 生产标准：P99 < 200ms                           │
├──────────────┼──────────────────────────────────────────────────┤
│ 并发数        │ Concurrent Users                                │
│              │ 同时在线并发送请求的用户数                        │
│              │ Little's Law: 并发数 = QPS × 平均RT              │
│              │ 例：QPS=1000, RT=0.1s → 并发数=100              │
├──────────────┼──────────────────────────────────────────────────┤
│ 错误率        │ Error Rate                                      │
│              │ 失败请求数 / 总请求数 × 100%                     │
│              │ 生产标准：< 0.1%                                 │
│              │ 压测标准：< 0.01%                                │
├──────────────┼──────────────────────────────────────────────────┤
│ 资源利用率    │ CPU / 内存 / 网络 / 磁盘IO                      │
│              │ 生产标准：< 80%（预留20%余量应对突发流量）        │
└──────────────┴──────────────────────────────────────────────────┘
```

### 3.2 Little's Law应用

```
Little's Law（利特尔法则）：
    L = λ × W

其中：
    L = 系统中平均请求数（并发数）
    λ = 请求到达速率（QPS）
    W = 平均响应时间（RT，单位秒）

应用示例：
    假设接口RT = 100ms = 0.1s
    目标QPS = 5000
    则需要的并发数 = 5000 × 0.1 = 500

    反过来：
    假设连接池最大连接数 = 200
    每个请求处理时间 = 50ms
    则最大QPS = 200 / 0.05 = 4000

这个公式用于：
1. 估算压测需要的并发连接数
2. 根据QPS目标推算系统资源需求
3. 验证压测结果是否合理（实测QPS是否接近理论值）
```

### 3.3 百分位数理解

```
为什么P99比平均值重要？

假设1000个请求的响应时间：
- 990个请求：50ms
- 9个请求：500ms
- 1个请求：5000ms

平均值 = (990×50 + 9×500 + 1×5000) / 1000 = 59.4ms
P99 = 500ms

平均值59.4ms看起来很好，但1%的用户（每1000个用户中有10个）
体验到了500ms以上的延迟，P99更真实地反映了用户体验。

百分位参考：
- P50（中位数）：50%的用户体验
- P90：90%的用户体验（关注长尾）
- P99：99%的用户体验（SLA标准）
- P999：99.9%的用户体验（极致体验，大厂标准）
```

## 压测策略

### 4.1 阶梯加压

```bash
#!/bin/bash
# step_pressure.sh: 阶梯加压测试
# 逐步增加并发量，找到系统的拐点（性能开始劣化的临界点）

JMETER="/opt/apache-jmeter-5.6.3/bin/jmeter"
TEST_PLAN="order_api_test.jmx"
REPORT_DIR="/data/test/reports/step_$(date +%Y%m%d_%H%M%S)"
mkdir -p $REPORT_DIR

# 压测梯度：50 → 100 → 200 → 500 → 1000 → 2000
LEVELS=(50 100 200 500 1000 2000)
DURATION=120  # 每个梯度持续120秒

for threads in "${LEVELS[@]}"; do
    echo "========================================"
    echo "开始压测: ${threads}并发, 持续${DURATION}秒"
    echo "时间: $(date '+%Y-%m-%d %H:%M:%S')"
    echo "========================================"
    
    $JMETER -n -t $TEST_PLAN \
        -l ${REPORT_DIR}/result_${threads}.jtl \
        -Jthreads=${threads} \
        -Jrampup=10 \
        -Jduration=${DURATION}
    
    echo "等待30秒让系统恢复..."
    sleep 30
done

echo "阶梯加压测试完成"
echo "结果目录: $REPORT_DIR"
```

```
阶梯加压预期结果示意：

QPS
  │
  │              ┌──── 拐点：超过此并发后QPS不再增长
  │           ╱──┘
  │        ╱
  │     ╱
  │  ╱
  │╱
  └────────────────────────→ 并发数
  50  100  200  500  1000  2000

RT
  │                          ┌── 拐点：RT开始急剧上升
  │                          │ ╱
  │                    ──────┘
  │                ╱
  │           ╱
  │      ╱
  │ ────
  └────────────────────────→ 并发数

通过阶梯加压找到拐点：
1. QPS不再增长的并发数 → 系统吞吐上限
2. RT开始急剧上升的并发数 → 响应时间拐点
3. 错误率开始出现的并发数 → 系统容量极限
```

### 5.2 CPU瓶颈分析


### 5.3 内存瓶颈分析

```bash
# 1. 查看系统内存
free -h
# 关注 available（可用内存）而非 free（包含buffer/cache）

# 2. 查看JVM内存使用
jmap -heap <java_pid>

# 3. GC频率分析
jstat -gcutil <java_pid> 1000 60
# 每秒采集一次，采集60秒
# 关注：
# - O (Old区使用率): 是否持续增长不回收 → 内存泄漏
# - YGC/YGCT: Young GC次数和总耗时
# - FGC/FGCT: Full GC次数和总耗时 → Full GC频繁说明需要调优

# 4. 内存泄漏排查
jmap -dump:format=b,file=heap.hprof <java_pid>
# 使用MAT分析：
# - Dominator Tree: 查看内存占用最大的对象
# - Leak Suspects: 自动分析内存泄漏嫌疑
# - Histogram: 按类统计对象数量和大小

# 5. JVM常见内存问题
# - Old区持续增长不回收 → 内存泄漏
# - Full GC频繁且耗时长 → 老年代空间不足
# - MetaSpace OOM → 动态类加载过多
# - Direct Memory OOM → Netty/NIO堆外内存泄漏
```

### 6.2 使用多台wrk压测

```bash
#!/bin/bash
# distributed_wrk.sh: 多机wrk分布式压测
# 通过SSH在多台机器上同时启动wrk

TARGET_URL="http://order-api.example.com/api/orders/10086"
SLAVES=("192.168.1.101" "192.168.1.102" "192.168.1.103")
THREADS=4
CONNS=500
DURATION=60

# 在所有Slave上同时启动wrk
for slave in "${SLAVES[@]}"; do
    ssh ${slave} "wrk -t${THREADS} -c${CONNS} -d${DURATION}s --latency ${TARGET_URL} > /tmp/wrk_${slave}.txt 2>&1" &
done

# 等待所有wrk完成
wait

# 收集结果
echo "=== 分布式压测结果汇总 ==="
TOTAL_QPS=0
for slave in "${SLAVES[@]}"; do
    QPS=$(ssh ${slave} "grep 'Requests/sec' /tmp/wrk_${slave}.txt | awk '{print \$2}'")
    echo "${slave}: QPS=${QPS}"
    TOTAL_QPS=$(echo "${TOTAL_QPS} + ${QPS}" | bc)
done
echo "=== 总QPS: ${TOTAL_QPS} ==="
```

### Q4：如何制定合理的性能目标

```
性能目标制定方法：

1. 基于历史数据
   - 统计日常峰值QPS的3~5倍作为压测目标
   - 大促场景按日常峰值的10~20倍

2. 基于业务场景
   - 读接口：P99 < 100ms，QPS目标 = 预估峰值 × 1.5
   - 写接口：P99 < 200ms，QPS目标 = 预估峰值 × 1.5
   - 查询接口：P99 < 500ms，支持异步化

3. 基于资源容量
   - 单机QPS = (CPU核数 × 1000) / 平均RT(ms)
   - 例：4核×10线程/核，RT=100ms → 单机QPS ≈ 40 × 1000/100 = 400
   - 总QPS = 单机QPS × 机器数 × 0.8（预留余量）

4. SLA标准
   - 可用性 > 99.99%
   - P99 RT < 目标值
   - 错误率 < 0.1%
   - 资源使用率 < 80%
```

> **总结：** 压测不是简单的「发请求看QPS」，而是一个系统工程——从工具选型、数据准备、场景设计、瓶颈定位到优化验证，每一步都需要严谨。核心原则是「模拟真实、逐步加压、精准定位、闭环验证」。压测的价值不在于发现系统能承受多大压力，而在于找到系统的极限在哪里、瓶颈是什么、如何在压力下保持稳定。
