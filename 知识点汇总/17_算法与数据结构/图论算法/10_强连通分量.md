def gabow_scc(graph):
    n = len(graph)
    indices = [-1] * n
    index = 0
    stack = []
    path_stack = []
    sccs = []

    def dfs(v):
        nonlocal index
        indices[v] = index
        index += 1
        stack.append(v)
        path_stack.append(v)
        in_component = False

        for w in graph[v]:
            if indices[w] == -1:
                dfs(w)
            elif indices[w] < indices[v]:

                scc.append(node)
                if node == v:
                    break

            dfs(v)

    return sccs
